موارد استفاده

مثال‌های عملی برای استفاده در برنامه‌ها، اسکریپت‌ها و ورک‌فلوها.

۱. ذخیره رمز عبور در برنامه‌ها

Python
from password_generator import generate
for name in ["ایمیل", "بانک", "شبکه_اجتماعی"]:
    print(f"{name}: {generate(length=20)}")

۲. عبارات عبور رمزنگاری

Python
from password_generator import generate_passphrase
phrase = generate_passphrase(words=6, capitalize=True, separator=" ")
print(f"کلید رمزنگاری: {phrase}")

۳. کدهای پشتیبان 2FA

Python
from password_generator import generate_pin
for i in range(10):
    print(f"کد {i+1}: {generate_pin(length=6, avoid_repeats=True)}")

۴. اطلاعات ورود DevOps

Python
from password_generator import generate, generate_pin
config = {
    "db_password": generate(length=32, exclude_ambiguous=True),
    "api_key": generate(length=48, symbols=False),
    "admin_pin": generate_pin(length=8, avoid_sequential=True),
}

۵. ممیزی قوت رمز عبور

Python
from password_generator import analyze
for pwd in ["password123", "Tr0ub4dor&3", "k8#Qx!2mNp@Lw9Rz"]:
    r = analyze(pwd)
    print(f"[امتیاز {r.score}] {r.feedback[0] if r.feedback else 'OK'}")

۶. خروجی مدیر رمز عبور

Python
from password_generator import generate
import json
entries = [{"service": s, "password": generate(length=20)} for s in ["github", "aws"]]
with open("passwords.json", "w") as f:
    json.dump(entries, f, indent=2)

۷. CI/CD

bash
export SECRET_KEY=$(python cli.py --length 48 --json | jq -r '.passwords[0]')

۸. ورک‌فلوی امن کلیپ‌بورد

Python
from password_generator import generate, analyze, copy_to_clipboard
pwd = generate(length=24)
if analyze(pwd).score >= 3:
    copy_to_clipboard(pwd, auto_clear_seconds=30)
