Use Cases

Real-world examples showing how to use Password Generator in your applications, scripts, and workflows.

1. Application Password Storage

Generate unique passwords for each account:

Python
from password_generator import generate

accounts = {
    "email": "user@gmail.com",
    "bank": "user@bank.com",
    "social": "user twitter",
    "cloud": "user aws",
}

for name, username in accounts.items():
    password = generate(length=20, symbols=True)
    print(f"{name} ({username}): {password}")
    # Save to your password manager!

2. Encryption Key Passphrases

Generate memorable passphrases for disk encryption or GPG keys:

Python
from password_generator import generate_passphrase

# LUKS disk encryption passphrase (6 words = ~66 bits)
phrase = generate_passphrase(words=6, capitalize=True, separator=" ")
print(f"Encryption passphrase: {phrase}")
# e.g. "Correct Horse Battery Staple Moon River"

# GPG key passphrase (8 words = ~88 bits)
gpg_pass = generate_passphrase(words=8, separator=".", capitalize=True)
print(f"GPG passphrase: {gpg_pass}")

4. DevOps Infrastructure Secrets

Generate credentials for databases, APIs, and services:

Python
from password_generator import generate, generate_pin
import json

# Generate infrastructure credentials
config = {
    "database": {
        "password": generate(length=32, exclude_ambiguous=True),
        "root_password": generate(length=48, symbols=False),
    },
    "api_keys": {
        "stripe": generate(length=48, uppercase=True, lowercase=True, digits=True, symbols=False),
        "sendgrid": generate(length=40),
    },
    "admin": {
        "pin": generate_pin(length=8, avoid_sequential=True),
        "ssh_key_passphrase": generate_passphrase(words=5, separator=" "),
    }
}

# Save to .env file
with open(".env", "w") as f:
    for section, keys in config.items():
        f.write(f"\n# {section.upper()}\n")
        for key, value in keys.items():
            f.write(f"{section.upper()}_{key.upper()}={value}\n")

print(json.dumps(config, indent=2))

5. Password Strength Audit

Audit existing passwords in your application:

Python
from password_generator import analyze, generate

# Test common passwords
test_passwords = [
    "password123",
    "Tr0ub4dor&3",
    "correct-horse-battery-staple",
    generate(length=20),
    generate(length=24, exclude_ambiguous=True),
]

print(f"{'Password':<35} {'Score':>5} {'Entropy':>8} {'Feedback'}")
print("-" * 80)

for pwd in test_passwords:
    report = analyze(pwd)
    masked = pwd[:20] + "..." if len(pwd) > 20 else pwd
    feedback = report.feedback[0] if report.feedback else "—"
    print(f"{masked:<35} {report.score:>5} {report.entropy:>6.0f}b {feedback}")

6. Password Manager Export

Export generated passwords as JSON for import into password managers:

Python
from password_generator import generate
import json

services = [
    {"name": "GitHub", "url": "github.com", "length": 24},
    {"name": "AWS", "url": "aws.amazon.com", "length": 32, "symbols": False},
    {"name": "Netflix", "url": "netflix.com", "length": 20},
    {"name": "Spotify", "url": "spotify.com", "length": 20},
]

entries = []
for service in services:
    entries.append({
        "name": service["name"],
        "url": f"https://{service['url']}",
        "username": f"your_username",
        "password": generate(
            length=service.get("length", 20),
            symbols=service.get("symbols", True),
        ),
    })

# Export
with open("passwords_export.json", "w") as f:
    json.dump(entries, f, indent=2)

print(f"Exported {len(entries)} passwords to passwords_export.json")

7. CI/CD Secret Generation

Generate secrets in deployment scripts:

bash
# Generate a secret and set as environment variable
export SECRET_KEY=$(python cli.py --length 48 --no-symbols --json | jq -r '.passwords[0]')

# Generate database password for Docker
DB_PASS=$(python cli.py --length 32 --json | jq -r '.passwords[0]')
docker run -e POSTGRES_PASSWORD=$DB_PASS postgres:15

# Generate and store in AWS Secrets Manager
aws secretsmanager create-secret \
    --name prod/db/password \
    --secret-string "$(python cli.py --length 40 --json | jq '.passwords[0]')"

8. Secure Clipboard Workflow

Generate, copy, and auto-clear for maximum security:

Python
from password_generator import generate, analyze, copy_to_clipboard
import time

def secure_password_workflow():
    """Generate and securely copy a password."""
    # Generate
    password = generate(length=24)

    # Analyze
    report = analyze(password)
    print(f"Generated password (score: {report.score}/4)")

    # Only copy strong passwords
    if report.score < 3:
        print("Password too weak, regenerating...")
        return secure_password_workflow()

    # Copy with 30-second auto-clear
    if copy_to_clipboard(password, auto_clear_seconds=30):
        print("Copied! Clipboard clears in 30 seconds.")
        print("Paste your password now!")
    else:
        print("Clipboard unavailable. Password:", password)

secure_password_workflow()

9. Custom Wordlist for Passphrases

Use a domain-specific wordlist:

Python
from password_generator import generate_passphrase, PassphraseConfig

# Create a themed wordlist (e.g., for a game)
with open("fantasy_words.txt", "w") as f:
    words = [
        "dragon", "castle", "wizard", "sword", "shield",
        "phoenix", "crystal", "shadow", "thunder", "flame",
        # ... at least 100 words
    ]
    f.write("\n".join(words))

# Generate a fantasy-themed passphrase
config = PassphraseConfig(
    words=4,
    separator="-",
    wordlist_path="fantasy_words.txt"
)
passphrase = generate_passphrase(config)
print(passphrase)  # e.g. "dragon-crystal-thunder-flame"

10. Bulk Generation with Reporting

Generate multiple passwords with a summary report:

Python
from password_generator import generate, analyze

def generate_batch(count=10, length=20):
    """Generate passwords with strength report."""
    results = []

    for i in range(count):
        pwd = generate(length=length)
        report = analyze(pwd)
        results.append({
            "password": pwd,
            "score": report.score,
            "entropy": report.entropy,
        })

    # Summary
    avg_score = sum(r["score"] for r in results) / len(results)
    avg_entropy = sum(r["entropy"] for r in results) / len(results)

    print(f"Generated {count} passwords")
    print(f"Average score: {avg_score:.1f}/4")
    print(f"Average entropy: {avg_entropy:.0f} bits")
    print(f"All score >= 3: {all(r['score'] >= 3 for r in results)}")

    return results

results = generate_batch()
